```js
#!/usr/bin/env node
'use strict';
// bin/generate-phase-spec.cjs
// SPEC-04: Translates task-envelope.json must_haves: truths: into per-phase TLA+ scratch spec.
// Each truth becomes INVARIANT (safety) or PROPERTY (liveness) stub.
//
// Usage:
//   node bin/generate-phase-spec.cjs .planning/phases/v0.21-04-spec-completeness/task-envelope.json
//   node bin/generate-phase-spec.cjs <envelope-path> --dry-run
//
// Output: formal/tla/scratch/<phase>-<timestamp>.tla

const fs = require('fs');
const path = require('path');

const ROOT = path.join(__dirname, '..');
const SCRATCH_DIR = path.join(ROOT, 'formal', 'tla', 'scratch');

// Classify a truth string as safety (INVARIANT) or liveness (PROPERTY)
// Safety keywords: threshold, depth, count, max, min, never, always, exactly, must, blocked, only
// Liveness keywords: eventually, deadline, async, progress, future, when, after, until, leads-to
function classifyTruth(truth) {
  const lower = truth.toLowerCase();
  const livenessKeywords = ['eventually', 'deadline', 'async', 'progress', 'future', 'when.*then', 'after', 'until', 'leads-to'];
  const isLiveness = livenessKeywords.some(kw => {
    if (kw.includes('.*')) return new RegExp(kw).test(lower);
    return lower.includes(kw);
  });
  return isLiveness ? 'PROPERTY' : 'INVARIANT';
}

// Generate TLA+ module content from task envelope
function generatePhaseSpec(taskEnvelope) {
  const phase = taskEnvelope.phase || 'unknown';
  const truths = taskEnvelope.plan?.must_haves?.truths || [];
  // Fallback: check top-level must_haves (some envelopes have flat structure)
  const topLevelTruths = taskEnvelope.must_haves?.truths || [];
  const allTruths = truths.length > 0 ? truths : topLevelTruths;

  // TLA+ module name: sanitize phase string
  const moduleName = 'Phase' + phase.replace(/[^a-zA-Z0-9]/g, '_') + 'Spec';
  const timestamp = new Date().toISOString();

  let spec = `---- MODULE ${moduleName} ----
(* Source: task-envelope.json must_haves: truths: *)
(* Generated by bin/generate-phase-spec.cjs — SPEC-04 *)
(* Phase: ${phase} *)
(* Generated: ${timestamp} *)
(* Purpose: Verify proposed state machine changes before quorum approval *)
(* PLACEHOLDER properties — developer fills in formal logic before TLC run *)

EXTENDS Naturals, FiniteSets, TLC

VARIABLES state, counter

TypeOK == state \\in {"INIT", "RUNNING", "DONE"}

Init == state = "INIT" /\\ counter = 0

Next ==
    \\/ /\\ state = "INIT" /\\ state' = "RUNNING" /\\ counter' = counter + 1
    \\/ /\\ state = "RUNNING" /\\ state' = "DONE" /\\ counter' = counter
    \\/ /\\ state = "DONE" /\\ UNCHANGED <<state, counter>>

Spec == Init /\\ [][Next]_<<state, counter>>

\\* ── Requirements as Properties ──────────────────────────────────

`;

  if (allTruths.length === 0) {
    spec += `\\* No must_haves: truths: found in task-envelope.json
\\* Add truths to see them translated to INVARIANT/PROPERTY stubs

`;
  }

  allTruths.forEach((truth, idx) => {
    const kind = classifyTruth(truth);
    const propName = `Req${String(idx + 1).padStart(2, '0')}`;
    // Truncate truth to 100 chars for comment; escape backslash
    const label = truth.substring(0, 100).replace(/\\/g, '\\\\').replace(/"/g, '\\"');
    spec += `${kind} ${propName} ==
  \\* "${label}${truth.length > 100 ? '...' : ''}"
  TRUE  \\* PLACEHOLDER: replace TRUE with formal TLA+ expression

`;
  });

  spec += `====`;
  return { moduleName, spec, truthCount: allTruths.length };
}

// CLI entrypoint
if (require.main === module) {
  const args = process.argv.slice(2).filter(a => !a.startsWith('--'));
  const flags = process.argv.slice(2).filter(a => a.startsWith('--'));
  const isDryRun = flags.includes('--dry-run');

  if (args.length === 0) {
    process.stderr.write('[generate-phase-spec] Usage: node bin/generate-phase-spec.cjs <task-envelope.json> [--dry-run]\n');
    process.exit(1);
  }

  const envelopePath = path.resolve(args[0]);
  if (!fs.existsSync(envelopePath)) {
    process.stderr.write('[generate-phase-spec] Error: task-envelope.json not found at ' + envelopePath + '\n');
    process.exit(1);
  }

  let envelope;
  try {
    envelope = JSON.parse(fs.readFileSync(envelopePath, 'utf8'));
  } catch (e) {
    process.stderr.write('[generate-phase-spec] Error: failed to parse task-envelope.json: ' + e.message + '\n');
    process.exit(1);
  }

  const { moduleName, spec, truthCount } = generatePhaseSpec(envelope);
  const ts = Date.now();
  const outName = (envelope.phase || 'unknown').replace(/[^a-zA-Z0-9-]/g, '_') + '-' + ts + '.tla';
  const outPath = path.join(SCRATCH_DIR, outName);

  if (isDryRun) {
    process.stdout.write('[generate-phase-spec] DRY-RUN: would write ' + outPath + '\n');
    process.stdout.write(spec + '\n');
  } else {
    fs.mkdirSync(SCRATCH_DIR, { recursive: true });
    fs.writeFileSync(outPath, spec, 'utf8');
    process.stdout.write('[generate-phase-spec] Written: ' + outPath + ' (' + truthCount + ' truths → INVARIANT/PROPERTY stubs)\n');
  }
}

module.exports = { generatePhaseSpec, classifyTruth };
```
