  <behavior>
evaluateTransitions(currentSnapshot, xstateEvent, machine):
  - Input: XState snapshot object (from actor.getSnapshot()), an XState event {type, ...payload}, the machine definition
  - Output: { currentState, event, possibleTransitions: [{target, guardName, guardPassed, guardContext}], expectedNextState, emptyTransitions }
  - If no transitions defined for this event+state: expectedNextState=null, emptyTransitions=true
  - If guard passes: guardPassed=true, that transition's target is expectedNextState
  - If guard fails: guardPassed=false, falls through to next transition or null

replayTrace(events, machine):
  - Input: array of XState events in sequence, machine definition
  - Output: array of {event, snapshotBefore, snapshotAfter, walkerResult} — full replay maintaining state across events
  - Starts from machine's initial state
  - Each event is sent to the actor; walker evaluates transitions before send, captures snapshot after
  - CRITICAL: uses ONE actor for the full sequence so context accumulated by event N-1 is visible to guards in event N

Cases:
  - quorum_start event in IDLE state → expected COLLECTING_VOTES (guard-free transition)
  - votes_collected in COLLECTING_VOTES, minQuorumMet=true → expected DELIBERATING
  - votes_collected in COLLECTING_VOTES, unanimityMet=true → expected DECIDED
  - decide event in DELIBERATING → expected DECIDED
  - unknown event type in any state → emptyTransitions=true, expectedNextState=null
  - multi-event sequence: QUORUM_START then VOTES_COLLECTED → second event is evaluated with context set by first (not from fresh IDLE actor)
  </behavior>

```javascript
'use strict';
// bin/xstate-trace-walker.cjs
// Reusable XState trace replay library.
// Evaluates machine transitions + guard conditions for a given event + snapshot.
// Used by validate-traces.cjs (TTrace export) and attribute-trace-divergence.cjs (root-cause attribution).

const path = require('path');
const fs   = require('fs');

function loadMachine() {
  const repoDist    = path.join(__dirname, '..', 'dist', 'machines', 'qgsd-workflow.machine.cjs');
  const installDist = path.join(__dirname, 'dist', 'machines', 'qgsd-workflow.machine.cjs');
  if (fs.existsSync(repoDist))    return require(repoDist);
  if (fs.existsSync(installDist)) return require(installDist);
  throw new Error('[xstate-trace-walker] Cannot find qgsd-workflow.machine.cjs');
}

// evaluateTransitions: given current snapshot, event, and machine, return transition analysis.
// Does NOT mutate snapshot — purely functional analysis.
function evaluateTransitions(currentSnapshot, xstateEvent, machine) {
  const stateName = typeof currentSnapshot.value === 'string'
    ? currentSnapshot.value
    : Object.keys(currentSnapshot.value)[0];
  const context = currentSnapshot.context || {};
  const stateConfig = machine.states?.[stateName];
  const transitions  = stateConfig?.on?.[xstateEvent.type];

  if (!transitions || (Array.isArray(transitions) && transitions.length === 0)) {
    return { currentState: stateName, event: xstateEvent, possibleTransitions: [], expectedNextState: null, emptyTransitions: true };
  }

  const transArr = Array.isArray(transitions) ? transitions : [transitions];
  const evaluated = transArr.map(trans => {
    const guardName  = trans.cond || trans.guard || null;
    const guardPassed = guardName ? evaluateGuard(guardName, context, xstateEvent, machine) : true;
    return { target: trans.target || stateName, guardName, guardPassed, guardContext: { ...context } };
  });

  const firstPass = evaluated.find(t => t.guardPassed);
  return {
    currentState: stateName,
    event: xstateEvent,
    possibleTransitions: evaluated,
    expectedNextState: firstPass ? firstPass.target : null,
    emptyTransitions: false,
  };
}

// evaluateGuard: look up guard function from machine's guards config and evaluate.
// Fail-open: returns true if guard cannot be resolved (avoids false divergences).
function evaluateGuard(guardName, context, event, machine) {
  try {
    // XState v4: guards defined in machine.options.guards or machine.config.guards
    const guards = machine.options?.guards || machine.config?.guards || {};
    const guardFn = guards[guardName];
    if (typeof guardFn !== 'function') return true; // unknown guard — fail-open
    return Boolean(guardFn(context, event));
  } catch (_) {
    return true; // guard threw — fail-open
  }
}

// replayTrace: replay an array of XState events maintaining state across events.
// Uses ONE actor for the entire sequence — context from event N-1 is visible to guards in event N.
// This is the key difference from validate-traces.cjs (which uses fresh actor per event — Pitfall 1).
function replayTrace(events, machine) {
  const { createActor } = require((function() {
    const repoDist    = path.join(__dirname, '..', 'dist', 'machines', 'qgsd-workflow.machine.cjs');
    const installDist = path.join(__dirname, 'dist', 'machines', 'qgsd-workflow.machine.cjs');
    return fs.existsSync(repoDist) ? repoDist : installDist;
  })());
  const actor = createActor(machine);
  actor.start();
  const results = [];
  for (const xstateEvent of events) {
    const snapshotBefore = actor.getSnapshot();
    const walkerResult   = evaluateTransitions(snapshotBefore, xstateEvent, machine);
    actor.send(xstateEvent);
    const snapshotAfter  = actor.getSnapshot();
    results.push({ event: xstateEvent, snapshotBefore, snapshotAfter, walkerResult });
  }
  actor.stop();
  return results;
}

module.exports = { evaluateTransitions, replayTrace, evaluateGuard, loadMachine };
```

**Implementation notes from research:**
- Guards are closures in XState v4 setup() — look in machine.options.guards (not machine.states[].on[].cond directly); the guard string is a name, the fn is in options.guards
- Pass context as a fresh copy `{ ...context }` to guardContext to avoid aliasing bugs (Pitfall 4)
- replayTrace starts fresh actor — state maintained via actor.send() sequence (fixes Pitfall 1 — multi-event interaction bugs)
- Fail-open on unknown guard: return true, not false (avoids false divergences from tool gaps)
- Do NOT modify any existing file; this is a new file only

  // This test guards against Pitfall 1 (fresh-actor validation blindspot):
  // If replayTrace creates a fresh actor per event, the second event would see IDLE context

  const events = [

    { type: 'VOTES_COLLECTED', successCount: 3 },
