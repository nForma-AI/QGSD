```javascript
// buildTTrace: construct a structured TTrace record for a divergent event.
// Uses xstate-trace-walker to get guard evaluation details.
// scoreboardMeta and confidence are passed from the outer loop.
// walker is the result of require('./xstate-trace-walker.cjs') — passed in to keep helper pure.
function buildTTrace(event, actualState, expectedState, divergenceType, scoreboardMeta, confidence, walker, machine) {
  let guardEvaluations = [];
  if (walker && machine && event) {
    try {
      // Build a snapshot representing the state BEFORE this event was sent.
      // We approximate this from the actualState since we're already past the event.
      // For a more precise snapshot, replayTrace in attribute-trace-divergence will replay from beginning.
      const { createActor } = require((function() {
        const p = require('path');
        const fs = require('fs');
        const repoDist    = p.join(__dirname, '..', 'dist', 'machines', 'qgsd-workflow.machine.cjs');
        const installDist = p.join(__dirname, 'dist', 'machines', 'qgsd-workflow.machine.cjs');
        return fs.existsSync(repoDist) ? repoDist : installDist;
      })());
      // Use a fresh actor to get a snapshot in the expected preceding state.
      // For COLLECTING_VOTES events, the actor was in COLLECTING_VOTES before diverging.
      const xstateEvt  = mapToXStateEvent(event);
      if (xstateEvt) {
        const actor = createActor(machine);
        actor.start();
        // Approximate: send QUORUM_START to reach COLLECTING_VOTES if needed
        if (expectedState === 'DELIBERATING' || expectedState === 'DECIDED') {
          actor.send({ type: 'QUORUM_START', slotsAvailable: event.slots_available || 3 });
        }
        const beforeSnap = actor.getSnapshot();
        actor.stop();
        const walkerResult = walker.evaluateTransitions(beforeSnap, xstateEvt, machine);
        guardEvaluations = walkerResult.possibleTransitions.map(t => ({
          guardName:    t.guardName || 'none',
          passed:       t.guardPassed,
          context:      t.guardContext,
        }));
      }
    } catch (_) { /* fail-open — guard evaluation failure must not break validation */ }
  }
  return {
    event,
    actualState,
    expectedState,
    guardEvaluations,
    divergenceType,
    confidence,
    observation_window: {
      n_rounds:    scoreboardMeta.n_rounds,
      window_days: scoreboardMeta.window_days,
    },
  };
}
```

```javascript
test('computeConfidenceTier: low confidence for 0 rounds', () => {

  assert.equal(computeConfidenceTier(10000, 90), 'high');

// ── buildTTrace ───────────────────────────────────────────────────────────────

test('buildTTrace: returns required TTrace fields', () => {
  const event = { action: 'quorum_complete', outcome: 'APPROVE', vote_result: 3, slots_available: 3 };
  const scoreboardMeta = { n_rounds: 5, window_days: 1 };
  const result = buildTTrace(event, 'DECIDED', 'DELIBERATING', 'state_mismatch', scoreboardMeta, 'low', null, null);
  assert.ok(result.event, 'event field required');
  assert.equal(result.actualState, 'DECIDED');
  assert.equal(result.expectedState, 'DELIBERATING');
  assert.equal(result.divergenceType, 'state_mismatch');
  assert.equal(result.confidence, 'low');
  assert.ok(result.observation_window, 'observation_window required');
  assert.ok(Array.isArray(result.guardEvaluations), 'guardEvaluations must be array');
});

test('buildTTrace: observation_window populated from scoreboardMeta', () => {
  const scoreboardMeta = { n_rounds: 42, window_days: 7 };
  const result = buildTTrace({}, 'A', 'B', 'state_mismatch', scoreboardMeta, 'medium', null, null);
  assert.equal(result.observation_window.n_rounds, 42);
  assert.equal(result.observation_window.window_days, 7);
});

test('buildTTrace: fail-open when walker is null — guardEvaluations is empty array', () => {
  const result = buildTTrace({ action: 'quorum_complete' }, 'DECIDED', 'DELIBERATING', 'state_mismatch',
    { n_rounds: 0, window_days: 0 }, 'low', null, null);
  assert.deepEqual(result.guardEvaluations, []);
});

test('validateMCPMetadata: non-mcp_call event returns true', () => {
  assert.equal(validateMCPMetadata({ action: 'quorum_start' }), true);
});
```
