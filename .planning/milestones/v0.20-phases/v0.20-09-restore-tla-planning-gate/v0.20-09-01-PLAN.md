# Read NDJSON output regardless of exit code (fail-open per PLAN-03)

  FV_FAILS=$(node -e "
    const fs = require('fs');
    try {
      const lines = fs.readFileSync('.formal/check-results.ndjson', 'utf8')
        .split('\n').filter(l => l.trim().length > 0);
      const fails = lines
        .map(l => { try { return JSON.parse(l); } catch(e) { return null; } })
        .filter(r => r && r.result === 'fail');
      if (fails.length > 0) {
        console.log(fails.map(r =>
          '- **' + r.check_id + '** (' + (r.surface || 'unknown') + '): ' + (r.summary || r.property || r.check_id)
        ).join('\n'));
      }
    } catch(e) { /* fail-open: NDJSON unreadable */ }
  " 2>/dev/null || echo "")

  if [ -n "$FV_FAILS" ]; then
    FV_STATUS="fail"
    FV_HYPOTHESES="$FV_FAILS"
    FV_GATE_CONTEXT="\n\n**FV Gate results — TLC failures detected (treat as hypotheses, not blockers per PLAN-03):**\n${FV_HYPOTHESES}\n\nThese TLA+ properties currently fail model checking. Consider whether the plans address these failures."
    echo "FV gate: $(echo "$FV_FAILS" | wc -l | tr -d ' ') TLC failure(s) detected — surfacing as hypotheses for quorum"
  else
    FV_STATUS="pass"
    echo "FV gate: no TLC failures — quorum proceeds normally"
  fi
else
  # NDJSON not found: TLC runner unavailable or run before first generate step
  echo "FV gate: check-results.ndjson not found — proceeding without FV context (PLAN-03 fail-open)"
fi
