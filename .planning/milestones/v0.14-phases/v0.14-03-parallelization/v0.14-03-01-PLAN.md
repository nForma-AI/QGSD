2. Replace the single `for (const step of steps)` loop (lines 255-270) with a new `runGroup(groupSteps)` async helper function and a two-phase execution:

```js
async function runGroup(groupSteps) {
  for (const step of groupSteps) {
    process.stdout.write(TAG + ' ' + SEP + '\n');
    process.stdout.write(TAG + ' [' + step.id + '] ' + step.label + '\n');
    process.stdout.write(TAG + ' ' + SEP + '\n');

    let passed = false;
    if (step.type === 'node') {
      passed = runNodeStep(step);
    } else if (step.type === 'wasm-dot') {
      passed = await runWasmDotStep(step);
    }

    const mark = passed ? '✓' : '✗';
    process.stdout.write('\n' + TAG + ' ' + mark + ' ' + step.id + '\n\n');
    record(step.id, step.label, passed);
  }
}
```

3. In the main IIFE, replace the `for..of` loop with:

```js
// ── Phase 1: Generate (must run first) ──────────────────────────────────────
const generateSteps = steps.filter(s => s.tool === 'generate');
const toolSteps     = steps.filter(s => s.tool !== 'generate');

if (generateSteps.length > 0) {
  process.stdout.write(TAG + ' Phase 1: Running generate steps sequentially...\n\n');
  await runGroup(generateSteps);
}

// ── Phase 2: Tool groups (concurrent) ───────────────────────────────────────
const toolGroupNames = [...new Set(toolSteps.map(s => s.tool))];

if (toolSteps.length > 0) {
  process.stdout.write(TAG + ' Phase 2: Running tool groups concurrently: ' + toolGroupNames.join(', ') + '\n\n');

  await Promise.all(
    toolGroupNames.map(tool => runGroup(toolSteps.filter(s => s.tool === tool)))
  );
}
```

4. **`--only` flag behavior**: When `--only=generate` is passed, `steps` contains only generate steps so `toolSteps` is empty and only Phase 1 runs. When `--only=tla`, `generateSteps` is empty (skipped) and Phase 2 runs with just the tla group. This preserves the existing `--only` semantics exactly.

5. **stdout interleaving note**: Since each `runGroup` call writes to `process.stdout.write` directly and steps within a group run sequentially, output within a group is serial. Groups running concurrently will interleave their output — this is acceptable for a CLI tool (common pattern in parallel build tools). Do NOT add mutex/lock mechanisms; the interleaving is intentional and readable.
