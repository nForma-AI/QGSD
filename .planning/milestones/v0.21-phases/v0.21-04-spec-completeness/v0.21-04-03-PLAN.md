**The Alloy spec must include:**

```alloy
-- .formal/alloy/quorum-composition.als
-- Source: REQUIREMENTS.md SPEC-03 + config.yaml composition rules
-- Verifies: no empty selection, high-risk full fan-out, solo mode single slot
-- SPEC-03: quorum composition selection rules

module quorum_composition

sig Config {
  riskLevel: one RiskLevel,
  soloMode: one Bool,
  availableSlots: set Slot,
  selectedSlots: set Slot,
  maxSize: one Int
} {
  maxSize > 0
  maxSize =< 5
  selectedSlots in availableSlots
}

abstract sig Slot {}
one sig Slot1, Slot2, Slot3, Slot4, Slot5 extends Slot {}

abstract sig RiskLevel {}
one sig Low, Medium, High extends RiskLevel {}

abstract sig Bool {}
one sig True, False extends Bool {}

-- SPEC-03 Rule 1: No empty selection when slots are available
fact NoEmptySelection {
  all c : Config |
    (#c.availableSlots > 0) implies (#c.selectedSlots > 0)
}

-- SPEC-03 Rule 2: High-risk configs use full fan-out (all available, up to maxSize)
fact HighRiskFullFanOut {
  all c : Config |
    (c.riskLevel = High) implies
      (#c.selectedSlots = min[#c.availableSlots, c.maxSize])
}

-- SPEC-03 Rule 3: Solo mode selects exactly one slot
fact SoloModeSingleSlot {
  all c : Config |
    (c.soloMode = True) implies (#c.selectedSlots = 1)
}

-- Satisfiability check: find instances within scope
run {} for 5 Config, 5 Slot, 4 int
```

**NOTE on Alloy syntax:** Alloy uses `--` for comments (not `(*`). `implies` is the implication operator. `=<` is less-than-or-equal. `min[a, b]` is the integer minimum. The `run` command scope `4 int` provides integers from -8 to 7 (needed for maxSize arithmetic).

If Alloy 6 `min[]` is not supported in this way, use an alternative: `c.riskLevel = High implies #c.selectedSlots >= #c.availableSlots` with a note explaining that this over-approximates when availableSlots > maxSize. Check .formal/alloy/quorum-votes.als for the Alloy integer syntax style used in this project.

**Also add an assertion for counterexample checking:**
```alloy
-- Assert: all rules must hold (Alloy checks this is unsatisfiable to find violations)
assert AllRulesHold {
  all c : Config | {
    (#c.availableSlots > 0) implies (#c.selectedSlots > 0)
    (c.riskLevel = High) implies (#c.selectedSlots > 0)
    (c.soloMode = True) implies (#c.selectedSlots = 1)
  }
}
check AllRulesHold for 5 Config, 5 Slot, 4 int
```
